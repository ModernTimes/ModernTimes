  2 /**

 10 class BuyItemAction extends CAction {
 11 
 12     /**
 13      * Does a couple of sanity checks
 14      * Then sellst he item and redirects to ShopAction
 15      * @param string $shopID int, but represented as string (bc of $GET)
 16      * @param string $itemID int, but represented as string (bc of $GET)
 17      * @param string $n int, but represented as string (bc of $GET)
 18      */
 19     public function run($shopID, $itemID, $n = "1") {
 20         /**
 21          * SYNTAX CHECKS
 22          */
 23 
 24         // shopID positive integer
 25         $validSyntax = (!empty($shopID)
 26                         && ctype_digit($shopID)
 27                         && $shopID > 0);
 28         if(!$validSyntax) {
 29             EUserFlash::setErrorMessage("Something went wrong. Shit happens.");
 30             $this->controller->redirect(array('index'));
 31         }
 32         
 33         // itemID and n positive integer
 34         $validSyntax = (!empty($itemID)
 35                         && ctype_digit($itemID)
 36                         && $itemID > 0
 37                         && !empty($n)
 38                         && ctype_digit($n)
 39                         && $n > 0);
 40         if(!$validSyntax) {
 41             EUserFlash::setErrorMessage("Something went wrong. Shit happens.");
 42             $this->controller->redirect(array('shop', 'shopID' => $shopID));
 43         }
 44 
 45         /**
 46          * SANITY CHECKS
 47          */
 48 
 49         // Does the Shop exist?
 50         $Shop = Shop::model()->with(array(
 51             'shopItems' => array(
 52                 'with' => array(
 53                     'item' => array(
 54                         'with' => array(
 55                             'charactermodifier'
 56                         )
 57                     )
 58                 )
 59             )
 60         ))->findByPk($shopID);
 61         if(!is_a($Shop, "Shop")) {
 62             EUserFlash::setErrorMessage("Something went wrong. Shit happens.");
 63             $this->controller->redirect(array('index'));
 64         }
 65 
 66         // Does the shop actually sell that item?
 67         // leaves $ShopItem at the right record
 68         $inStock = false;
 69         foreach($Shop->shopItems as $ShopItem) {
 70             if($ShopItem->item->id == $itemID) {
 71                 $inStock = true;
 72                 break;
 73             }
 74         }
 75         if(!$inStock) {
 76             EUserFlash::setErrorMessage("Something went wrong. Shit happens.");
 77             $this->controller->redirect(array('shop', 'shopID' => $shopID));
 78         }
 79 
 80         // Is the Character allowed to enter that shop?
 81         $Character = CD();
 82         if(!$Shop->call("meetsRequirements", $Character)) {
 83             // RequirementCheckerBehavior generates messages
 84             $this->controller->redirect(array('index'));
 85         }
 86         
 87         // Does the Character have enough resources to buy the stuff?
 88         if($Character->cash < $ShopItem->cash * $n ||
 89                 $Character->favours < $ShopItem->favours * $n ||
 90                 $Character->kudos < $ShopItem->kudos * $n) {
 91             
 92             EUserFlash::setErrorMessage("You don't have enough resources to buy this.");
 93             $this->controller->redirect(array('shop', 'shopID' => $shopID));
 94         }
 95         
 96         /**
 97          * BEWARE: ACTUAL BUSINESS LOGIC 
 98          */
 99         
100         $transaction = Yii::app()->tools->getTransaction();
101         try {
102             $Character->gainItem($ShopItem->item, $n);
103             $Character->decreaseCash($ShopItem->cash * $n);
104             $Character->decreaseFavours($ShopItem->favours * $n);
105             $Character->decreaseKudos($ShopItem->kudos * $n);
106             $Character->update();
107             $transaction->commit();
108        } catch(Exception $e) {
109             $transaction->rollback();
110             EUserFlash::setErrorMessage("Weird database shit happened.");
111         }
112                 
113         $this->controller->render("shop", array(
114             "Shop" => $Shop,
115         ));
116     }
