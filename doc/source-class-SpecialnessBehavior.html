 3 /**
 4  * Makes it possible to define special behavior for specific entries for
 5  * certain models, like special monsters, items, areas, etc.
 6  * 
 7  * Important: If a model can have specially defined behaviors, do ot call that 
 8  * model's function directly. Use the $model->call('functionName', params) 
 9  * syntax instead. SpecialnessBehavior will then redirect the call to either 
10  * the standard model's function or to the special function of a class defined 
11  * by the attribute specialClass in the model entry.
12  */
13 
14 class SpecialnessBehavior extends CBehavior {
15 
16     /**
17      * Non-magic call function
18      * Redirects requests to the special method, if it exists, and to the basic
19      * method of the owner class otherwise
20      * @param string 1st, name of the function to be called
21      * @params mixed 2nd+, whatever params you want to use
22      * @return mixed, whatever the called function returns
23      */
24     public function call() {
25         $name = func_get_arg(0);
26         $parameters = array();
27         
28         // ToDo: more elegant!?!
29         $numArgs = func_num_args();
30         for($i = 1; $i < $numArgs; $i++) {
31             $parameters[] = func_get_arg($i);
32         }
33     
34         /**
35          * if specialnessBehavior method exists, call that
36          * if not, or if something goes wrong, fall through to basic method
37          */
38         if($this->isSpecial()) {
39             $this->initSpecialness();
40             if(method_exists($this->owner->asa("special"), $name)) {
41                 /* 
42                 Yii::trace("calling " . get_class($this->owner) . "." . $name
43                     . (isset($this->owner->name) 
44                         ? " (" . $this->owner->name . ") " : "") 
45                     . " - SPECIAL");
46                  */
47                 return call_user_func_array(array($this->owner->asa("special"), $name), $parameters);
48             }
49         }
50         /* 
51          Yii::trace("calling " . get_class($this->owner) . "." . $name
52             . (isset($this->owner->name) 
53                 ? " (" . $this->owner->name . ") " : "") 
54             . " - BASIC");
55          */
56         return call_user_func_array(array($this->owner, $name), $parameters);
57     }
58 
59     // ToDo: Do some checking for classname, is_child_of behavior, etc.
60     public function initSpecialness() {
61         if($this->isSpecial() && $this->owner->asa("special") === null) {
62             $this->owner->attachbehavior('special', new $this->owner->specialClass);
63 
64             Yii::trace("attached special behavior class " . $this->owner->specialClass . " to " . get_class($this->owner) . (isset($this->owner->name) 
65                     ? " (" . $this->owner->name . ") " : ""));
66         }
67     }
68     public function isSpecial() {
69         return (!empty($this->owner->specialClass) ? true : false);
70     }
71 }
