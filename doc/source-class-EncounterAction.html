 34     /**
 35      * Checks if the Encounter is legitimate and renders the encounter view 
 36      */
 37     public function run() {
 38         $Character = CD();
 39         /**
 40          * Encounters can't be requested by the user directly, at least not 
 41          * the first step of an encounter path
 42          * @todo Syntax checks
 43          */
 44         if($Character->ongoingEncounterID == null && !$this->callFromWithinApplication) {
 45             EUserFlash::setErrorMessage("Invalid choice", 'validate');
 46             $this->controller->redirect('index');
 47         }
 48         
 49         if(empty($this->encounterID) && empty($this->encounter)) {
 50             if(!empty($_GET['encounterID'])) {
 51                 // @todo syntax checks
 52                 $this->encounterID = (int) $_GET['encounterID'];
 53             } elseif (!empty(CD()->ongoingEncounterID)) {
 54                 $this->encounterID = CD()->ongoingEncounterID;
 55             }
 56         }
 57         
 58         if(empty($this->encounter)) {
 59             $this->encounter = Encounter::model()->withRelated()->findByPk($this->encounterID);
 60         }
 61         
 62         if(!is_a($this->encounter, "Encounter")) {
 63             // exception
 64         }
 65 
 66         // legitimate = can the player arrive at this encounter, given the current
 67         // and previous encounters?
 68         if(!$this->legitimateEncounter($Character)) {
 69             EUserFlash::setErrorMessage("Invalid choice", 'validate');
 70             Yii::app()->controller->redirect(array('game/encounter', array('encounterID' => CD()->ongoingEncounterID)));
 71         }
 72         
 73         
 74         // Resolve the encounter
 75         // d($this->encounter);
 76         
 77         $this->encounter->run($Character);
 78         
 79         $this->controller->layout = '//layouts/g_sidebar';
 80         $this->controller->render('encounter', array("encounter" => $this->encounter));
 81     }
 82 
 83     /**
 84      * Checks if $this->encounter is legitimate, i.e. if the Character can
 85      * actually be at this point in an Encounter path
 86      * @param Character $Character
 87      * @return boolean 
 88      */
 89     public function legitimateEncounter($Character) {
 90         // No ongoing encounter = fine
 91         if($Character->ongoingEncounterID != null) {
 92             // current encounter = to_id from last encounter = fine
 93             // encounterEncounters instead?
 94             foreach($this->encounter->encounterEncounters1 as $previousEncounter) {
 95                 if($previousEncounter->fromEncounterID == $Character->ongoingEncounterID) {
 96                     return true;
 97                 }
 98             }
 99             // No previous encounter with to_id = current encounter = bad
100             return false;
101         }
102         return true;
103     }
