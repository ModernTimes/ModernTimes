 3 /**
 4  * "My first deck": The character is sent to an Undisclosed insurance company
 5  * to create their first professional PowerPoint deck.
 6  * 
 7  * Upon completion, the Character gets a cash bonus.
 8  * 
 9  * This Quest is automatically added for new consultants with state = available.

11  * @uses SpecialnessBehavior
12  * @uses CharacterQuests
13  * @package Quests

15 
16 class Consultant1Quest extends CBehavior {
17 
18     /**
19      * Amount of cash that the character gets for completing the quest
20      * @constant int 
21      */
22     const rewardCash = 2000;
23     
24     /**
25      * ID of the required item
26      * @constant int 
27      */
28     const deckItemID = 19;
29     
30     /**
31      * Attaches custom event handlers to a Character
32      * @param Character $Character 
33      */
34     public function attachToCharacter($Character) {
35         $Character->onGainItem = array($this, 'reactToOnGainItem');
36     }
37     
38     /**
39      * Detaches custom event handlers from a Character
40      * @param Character $Character 
41      */
42     public function detachFromCharacter($Character) {
43         $Character->detachEventHandler("onGainItem", array($this, 'reactToOnGainItem'));
44     }
45     
46     /**
47      * Checks whether the character got a novice deck
48      * @param GainItemEvent $event 
49      */
50     public function reactToOnGainItem($event) {
51         if($event->getItem()->id == self::deckItemID) {
52             $this->owner->setState("succeeded");
53         }
54     }
55 
56     /**
57      * Generates a user message
58      * @param QuestChangeStateEvent $event 
59      */
60     public function reactToOnOngoing($event) {
61         EUserFlash::setSuccessMessage("You agreed to make your very own PowerPoint presentation. Your client, an undisclosed insurance company, is marked on your map.");
62     }
63     
64     /**
65      * Generates a user message
66      * @param QuestChangeStateEvent $event 
67      */
68     public function reactToOnSucceeded($event) {
69         EUserFlash::setSuccessMessage("Wow -- you actually made your very own first PowerPoint presentation. Make sure to tell your boss about it!");
70     }
71     
72     /**
73      * - Gives out the reward
74      * - Sets the CharacterQuest to visible
75      * @param QuestChangeStateEvent $event 
76      */
77     public function reactToOnCompleted($event) {
78         EUserFlash::setSuccessMessage("Nice presentation, little grasshopper! Don't forget to equip your new weapon when you go to your next client.");
79         
80         $this->owner->Character->gainCash(self::rewardCash, "quest");
81         $this->owner->Character->update();
82         
83         // Reset params and update CharacterQuest record
