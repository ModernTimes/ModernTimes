44         foreach($CharacterContact->statuses as $status) {

46                 case 1:
47                     if(!$CharacterContact->{$status}) {
48                         if($generateMessages) {
49                             EUserFlash::setErrorMessage($CharacterContact->name . " needs to be " . $status . " to do that");
50                         }
51                         $meetsRequirements = false;
52                     }
53                     break;
54                 case -1:
55                     if($CharacterContact->{$status}) {
56                         if($generateMessages) {
57                             EUserFlash::setErrorMessage($CharacterContact->name . " must not be " . $status);
58                         }
59                         $meetsRequirements = false;
60                     }
61                 default:
62                     break;
63             }
64         }
65
