  4  * Tutorial quest

 33     /**
 34      * Set the initial parameters for the Quest.
 35      * In this case: flags for each step

 37     public function setInitialParams() {
 38             $this->owner->params['currentStep'] = 'overview';
 39             $this->owner->params['inventory'] = 0;
 40             $this->owner->params['prepInventory'] = 0;
 41             $this->owner->params['skills'] = 0;
 42             $this->owner->params['prepSkills'] = 0;
 43             $this->owner->params['battleskills'] = 0;
 44             $this->owner->saveParams();

 47     /**
 48      * Returns a string representation of what is going on right now with
 49      * regard to this Quest
 50      * @return string
 51      */
 52     public function getDescStatus() {
 53         if(!empty($this->owner->params['currentStep'])) {
 54             switch($this->owner->params['currentStep']) {
 55                 case "inventory":
 56                     return "<BR />You remember having some things in your closet. Click on 'Stuff' in the navigation bar to have a closer look. You should probably put on some of the things you find there. Unless you want to paint the town nakedly.";
 57                     break;
 58                 case "skills":
 59                     return "<BR />You do have some talents, now that you think about it. Click on 'Skills' in the navigation bar, then click on your talent to see what happens.";
 60                     break;
 61                 case "battleskills":
 62                     return "<BR /><span class='label label-important'>KNOCK</span> <span class='label label-important'>KNOCK</span> Seems like you have other worries than protecting yourself. Better pick yourself up and open the door. Oh, by the way: <button class='btn btn-mini'><i class='icon-time'></i></button> indicates that opening the door will cost you 1 turn. Every day, you will get 50 turns to spend as you please. You can see the number of remaining turns under your character's profile on the left.";
 63                     break;
 64                 default:
 65                     return "";
 66                     break;
 67             }
 68         } else {
 69             return "";
 70         }

 73     /**
 74      * Set the current step to $step
 75      * @param string $step enum(inventory|skills|etc)
 76      */
 77     public function startStep($step) {
 78         $this->owner->params['currentStep'] = $step;
 79     }
 80     /**
 81      * Sets params[step] = 1 and currentStep back to 'overview'
 82      * @param string $step enum(inventory|skills|etc)
 83      * @return bool true if step wasn't completed before, false otherwise
 84      */
 85     public function completeStep($step) {
 86         $stepBefore = $this->owner->params[$step];
 87         $this->owner->params[$step] = 1;
 88         $this->owner->params['currentStep'] = 'overview';
 89         return (!$stepBefore);
 90     }

 93     /**
 94      * Prepares the inventory step, if it has not been prepared before
 95      * Gives out initial items.
 96      * Does NOT call CharacterQuest->quest->saveParams !!
 97      */
 98     public function prepareStepInventory() {
 99         if(!$this->owner->CharacterQuest->quest->params["prepInventory"]) {
100             $Item1 = Item::model()->findByPk(1);
101             $Item2 = Item::model()->findByPk(2);
102             $Item3 = Item::model()->findByPk(3);
103             $this->owner->Character->gainItem($Item1, 1, "quest");
104             $this->owner->Character->gainItem($Item2, 1, "quest");
105             $this->owner->Character->gainItem($Item3, 1, "quest");
106             $this->owner->CharacterQuest->quest->params["prepInventory"] = 1;
107         }
108     }

110     /**
111      * Prepares the skills step, if it hasn't been prepared before
112      * Gives out initial skill
113      * Does NOT call CharacterQuest->quest->saveParams !!
114      */
115     public function prepareStepSkills() {
116         if(!$this->owner->CharacterQuest->quest->params["prepSkills"]) {
117             $StartCharacterSkill = $this->owner->Character->getCharacterSkill(1);
118             $StartCharacterSkill->available = 1;
119             $StartCharacterSkill->save();
120             $this->owner->CharacterQuest->quest->params["prepSkills"] = 1;
121         }
122     }

149     /**
150      * Checks whether the character equipped one of the starting items
151      * @uses completeStep
152      * @uses Quest->saveParams
153      * @param EquipItemEvent $event 
154      */
155     public function reactToOnEquipItem($event) {
156         if(!$this->owner->params['inventory']) {
157             $Item = $event->getItem();
158             $startingItems = array(1,2,3);
159             if(in_array($Item->id, $startingItems)) {
160                 if($this->completeStep("inventory")) {
161                     EUserFlash::setSuccessMessage("Congratulations! You found your things.");
162                     if($this->isFinished()) {
163                         $this->owner->setState("completed");
164                     } else {
165                         $this->owner->saveParams();
166                     }
167                 }
168             }
169         }
170     }

173      * Checks whether the character used their basic non-combat skill
174      * @uses completeStep
175      * @uses Quest->saveParams
176      * @param UseSkillEvent $event

178     public function reactToOnUseSkill($event) {
179         if(!$this->owner->params['skills']) {
180             $Skill = $event->getSkill();
181             $startingSkills = array(1);
182             if(in_array($Skill->id, $startingSkills)) {
183                 if($this->completeStep("skills")) {
184                     EUserFlash::setSuccessMessage("Yeah, that's it. Did you notice that '" . $Skill->name . "' gave you an effect that makes you more powerful? Effects are displayed under your character's profile on the left. The little number next to the effect's name indicates how many turns the effect will last.  You can move your mouse over the effect's name to get further information.");
185                     if($this->isFinished()) {
186                         $this->owner->setState("completed");
187                     } else {
188                         $this->owner->saveParams();

191             }

195     /**
196      * Checks if the character killed the spider under their bed
197      * @uses completeStep
198      * @uses Quest->saveParams
199      * @param UseSkillEvent $event 
200      */
201     public function reactToOnFinishedBattle($event) {
202         if(!$this->owner->params['battleskills']) {
203             $Battle = $event->getBattle();
204             if($Battle->isWinner($this->owner->Character) &&
205                     is_a($Battle->combatantB, "Monster") &&
206                     $Battle->combatantB->id == self::monsterID) {
207                 
208                 if($this->completeStep("battleskills")) {
209                     EUserFlash::setSuccessMessage("Great job! Look like you're not as groggy as it seemed.");
210                     if($this->isFinished()) {
211                         $this->owner->setState("completed");
212                     } else {
213                         $this->owner->saveParams();
214                     }
215                 }
216             }
217         }
218     }

220     /**
221      * Gives out all the rewards that the player hasn't collected yet
222      * @uses prepareStepInventory
223      * @uses prepareStepSkills
224      * @uses prepareStepBattleskills
225      * @param QuestChangeStateEvent $event 
226      */
227     public function reactToOnCompleted($event) {
228         if($this->isFinished()) {
229             EUserFlash::setSuccessMessage("Congratulations! You woke up successfully. Now that you're sober, you even remember that you have an appointment with your employer. The location is marked on your map. Just click on 'London' in the navigation bar to check it out.");
230         } else {
231             foreach($this->_steps as $step) {
232                 call_user_func(array($this, "prepareStep" . ucfirst($step)));
233             }
234 
235             EUserFlash::setSuccessMessage("Right. You're alive, and whatever you can't remember right now will surely come back to you when you need it. Actually, something IS coming back to you right now, namely the appointment you have with your employer. The location is marked on your map. Just click on 'London' in the navigation bar to check it out.");
236             
237             if($this->isFinished()) {
238                 $this->owner->setState("completed");
239             }
240         }
241         
242         // Reset params and update CharacterQuest record
243         $this->owner->reactToOnCompleted($event);
244     }

246     /**
247      * Checks if all the steps in the quest have been completed
248      * @return boolean 
249      */
250     public function isFinished() {
251         foreach($this->_steps as $step) {
252             if(!$this->owner->CharacterQuest->quest->params[$step]) {
253                 return false;
254             }
255         }
256         return true;
257     }
