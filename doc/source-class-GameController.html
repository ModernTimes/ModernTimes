123     /**
124      * - Add the GoodForNothing Effect if Character has no hp left
125      * - Save the Character record in case it has changed during the action
126      * @param CAction $action
127      * @return boolean always true
128      */
129     public function afterAction($action) {
130         $character = CD();
131 
132         // Add GoodForNothing effect if character has 0 hp
133         if($character->hp <= 0) {
134             // Add GFN (ID 1) for 3 turns, unless it's already there
135             Yii::app()->tools->addEffect(1, 3, array('addTurns' => false));
136         }
137         
138         // d($character);
139         // Save Character AR in case it changed
140         // (whether or not AR has changed is checked by AR behavior automatically)
141         Yii::app()->cd->save();
142 
143         Yii::endProfile("game action: " . $action->id);
144         
145         return true;
146     }
