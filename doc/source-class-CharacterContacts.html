  5 /**
  6  * Basic HAS_MANY association model
  7  * Which contacts does the character have?
  8  * 
  9  * See BaseCharacterContacts for a list of attributes and related Models
 10  * 
 11  * @see Character
 12  * @see Contact
 13  * @package Character.Relations
 14  */
 15 
 16 class CharacterContacts extends BaseCharacterContacts {
 17     
 18     /**
 19      * array with string representations of potential treatments of contacts
 20      * @var array
 21      */
 22     public $treatments = array("befriendable", "bribable", "seducible");
 23     
 24     /**
 25      * array with string representations of statuses that a contact can be in
 26      * @var array
 27      */
 28     public $statuses = array("befriended", "bribed", "seduced");
 29     
 30     /**
 31      * Checks whether the character is treated in some way already
 32      * @return boolean 
 33      */
 34     public function isTreated() {
 35         foreach($this->statuses as $status) {
 36             if($this->{$status}) {
 37                 return true;
 38             }
 39         }
 40         return false;
 41     }
 42     
 43     /**
 44      * Returns the status of a treated contact
 45      * @return boolean 
 46      */
 47     public function getStatus() {
 48         foreach($this->statuses as $status) {
 49             if($this->{$status}) {
 50                 return $status;
 51             }
 52         }
 53         return "untreated";
 54     }
 55     
 56     /**
 57      * Returns an array with string identifiers of possible treatments
 58      * @return array
 59      */
 60     public function getPossibleTreatments() {
 61         $possibleTreatments = array();
 62         foreach($this->treatments as $treatment) {
 63             if($this->{$treatment}) {
 64                 $possibleTreatments[] = $treatment;
 65             }
 66         }
 67         return $possibleTreatments;
 68     }
 69     
 70     /**
 71      * Returns a description of the status of this contact
 72      * @uses getStatus
 73      * @return string 
 74      */
 75     public function getStatusDesc() {
 76         $status = $this->getStatus();
 77         switch($status) {
 78             case "befriended":
 79                 return "You guys are <b>friends</b>";
 80             case "bribed":
 81                 return "You <b>bribed</b> " . _objective($this->sex);
 82             case "seduced":
 83                 return "You <b>seduced</b> " . _objective($this->sex);
 84             // should never happen
 85             case "untreated":
 86                 return "You don't really know " . _objective($this->sex) . " yet";
 87             // should never happen
 88             default:
 89                 return $this->name . " is <b>" . $status . "</b>";
 90         }
 91     }
 92     
 93     /**
 94      * Generate a bye-bye message regarding loosing this CharacterContact 
 95      */
 96     public function byebye() {
 97         EUserFlash::setSuccessMessage(
 98             "You lost your grip over " . $this->name . ". " . 
 99             "But who cares? " . ucfirst(_personal($this->sex)) .
100             " fulfilled " . _possessive($this->sex) . " purpose."
101         );
102     }
