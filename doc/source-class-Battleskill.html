  6 /**
  7  * Resolve the battleskill by interfering with the battle object

  9  * hero and enemy are from the Skill user's point of view

 17 class Battleskill extends BaseBattleskill {
 18 
 19     /**
 20      * Other Skills or Battleeffects can block this skill, which means that
 21      * it's not executed
 22      * @var bool
 23      */
 24     public $blocked = false;
 25 
 26     /**
 27      * Resolves the Battleskill by calling other functions to take care of basic
 28      * Skill mechanics
 29      * @uses checkBlocked
 30      * @uses dealDamage
 31      * @uses createEffects
 32      * @param Battle $battle
 33      * @param CombatantBehavior $hero Model record with CombatantBehavior
 34      * @param CombatantBehavior $enemy Model record with CombatantBehavior
 35      */
 36     public function resolve($battle, $hero, $enemy) {
 37         if(!$this->call("checkBlocked", $battle, $hero, $enemy)) {
 38             return;
 39         }
 40         
 41         $this->call("dealDamage", $battle, $hero, $enemy);
 42         $this->call("createEffects", $battle, $hero, $enemy);
 43     }
 44     
 45     /**
 46      * Checks if this Battleskill is blocked and if so adds a Battlemessage
 47      * @param Battle $battle
 48      * @param CombatantBehavior $hero Model record with CombatantBehavior
 49      * @param CombatantBehavior $enemy Model record with CombatantBehavior
 50      * @return bool not blocked: true, blocked: false
 51      */
 52     public function checkBlocked($battle, $hero, $enemy) {
 53         if($this->blocked) {
 54             $battleMsg = new Battlemessage("", $this);
 55             $battleMsg->setResult("blocked");
 56             $battle->log($hero, $battleMsg);
 57             
 58             /**
 59              * Remember that blocked is a property of the skill class and gets
 60              * carried over to future rounds unless it is reset properly
 61              */
 62             $this->blocked = false;
 63             
 64             return false;
 65         }
 66         return true;
 67     }

 69     /**
 70      * Basic Battleeffect creation
 71      * @param Battle $battle
 72      * @param CombatantBehavior $hero Model record with CombatantBehavior
 73      * @param CombatantBehavior $enemy Model record with CombatantBehavior
 74      * @param bool $log whether the effect creation should be looged with a 
 75      * Battlemessage
 76      */
 77     public function createEffects($battle, $hero, $enemy, $log = true) {
 78         if($this->createBattleeffectID != null) {
 79             $Battleeffect = clone $this->createBattleeffect;
 80             
 81             // See Battleeffect->initialize for further details
 82             $result = $Battleeffect->call("initialize", $battle, $hero, $enemy, array(
 83                 'turns' => $this->battleeffectTurns)
 84             );
 85             if($log) {
 86                 $battleMsg = new Battlemessage("", $this);
 87                 if($result == "added") {
 88                     $battleMsg->msg = sprintf($this->call("getMsgResolved"), $hero->name);
 89                 } elseif($result == "increasedDuration") {
 90                     $battleMsg->msg = sprintf($this->call("getMsgIncreasedDuration"), $hero->name);
 91                 }
 92                 $battleMsg->setResult("effect", $Battleeffect);
 93                 $battle->log($hero, $battleMsg);
 94             }
 95         }
 96     }

 98     /**
 99      * Basic damage dealing mechanics
100      * @param Battle $battle
101      * @param CombatantBehavior $hero Model record with CombatantBehavior
102      * @param CombatantBehavior $enemy Model record with CombatantBehavior
103      * @uses onBeforeDealingDamage
104      * @uses onAfterDealingDamage
105      * @uses BattleActionDamageEvent
106      * @uses BattleActionDamageDealtEvent
107      */
108     public function dealDamage($battle, $hero, $enemy) {
109         if($this->dealsDamage) {
110             $damage = $this->damageFixedAmount;
111             $damageType = $this->damageType;
112 
113             if ($this->costEnergy == 0) {
114                 $damage += $this->damageAttackFactor * $hero->getNormalAttack();
115             } else {
116                 $damage += $this->damageAttackFactor * $hero->getSpecialAttack();
117             }
118 
119             // Bonus collection
120             $event = new BattleActionDamageEvent($battle, $hero, $enemy, $this, 
121                     $damage, $damageType);
122             $battle->onBeforeDealingDamage($event);
123 
124             $damageAdjusted = max(0, floor($event->adjustStat($damage)));
125             $damageDone = $enemy->takeDamage($damageAdjusted, $damageType);
126 
127             $battleMsg = new Battlemessage(sprintf($this->call("getMsgResolved"), $hero->name), $this);
128             $battleMsg->setResult("damage", $damageDone, $damageType);
129             $battle->log($hero, $battleMsg);
130             
131             // Notification only
132             $event = new BattleActionDamageDealtEvent($battle, 
133                     $hero, $enemy, $this, 
134                     $damageDone, $damageType
135             );
136             $battle->onAfterDealingDamage($event);
137         }
138     }
