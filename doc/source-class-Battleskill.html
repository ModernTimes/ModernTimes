 19     /**
 20      * Other Skills or Battleeffects can block this skill, which means that
 21      * it's not executed
 22      * @var bool
 23      */
 24     public $blocked = false;

 26     /**
 27      * Resolves the Battleskill by calling other functions to take care of basic
 28      * Skill mechanics
 29      * @uses checkBlocked
 30      * @uses checkSuccess
 31      * @uses dealDamage
 32      * @uses createEffects
 33      * @param Battle $battle
 34      * @param CombatantBehavior $hero Model record with CombatantBehavior
 35      * @param CombatantBehavior $enemy Model record with CombatantBehavior
 36      */
 37     public function resolve($battle, $hero, $enemy) {
 38         if(!$this->call("checkBlocked", $battle, $hero, $enemy)) {
 39             return;
 40         }
 41         if(!$this->call("checkSuccess", $battle, $hero, $enemy)) {
 42             return;
 43         }
 44 
 45         if($this->costEnergy > 0) {
 46             $hero->decreaseEnergy($this->costEnergy);
 47         }
 48         
 49         $this->call("dealDamage", $battle, $hero, $enemy);
 50         $this->call("createEffects", $battle, $hero, $enemy);
 51     }

 53     /**
 54      * Checks if this Battleskill is blocked and if so adds a Battlemessage
 55      * @param Battle $battle
 56      * @param CombatantBehavior $hero Model record with CombatantBehavior
 57      * @param CombatantBehavior $enemy Model record with CombatantBehavior
 58      * @return bool not blocked: true, blocked: false
 59      */
 60     public function checkBlocked($battle, $hero, $enemy) {
 61         if($this->blocked) {
 62             $battleMsg = new Battlemessage("", $this);
 63             $battleMsg->setResult("blocked");
 64             $battle->log($hero, $battleMsg);
 65             
 66             /**
 67              * Remember that blocked is a property of the skill class and gets
 68              * carried over to future rounds unless it is reset properly
 69              */
 70             $this->blocked = false;
 71             
 72             return false;
 73         }
 74         return true;
 75     }

 77     /**
 78      * Checks if this Battleskill is used successfully 
 79      * (RNG based on successRate)
 80      * @param Battle $battle
 81      * @param CombatantBehavior $hero Model record with CombatantBehavior
 82      * @param CombatantBehavior $enemy Model record with CombatantBehavior
 83      * @return bool success: true, failure: false
 84      */
 85     public function checkSuccess($battle, $hero, $enemy) {
 86         if($this->successRate == 1) {
 87             return true;
 88         } else {
 89             if(mt_rand(0, 100) > $this->successRate * 100) {
 90                 $battleMsg = new Battlemessage("", $this);
 91                 $battleMsg->setResult("failed");
 92                 $battle->log($hero, $battleMsg);
 93                 return false;
 94             } else {
 95                 return true;
 96             }
 97         }
 98     }

100     /**
101      * Basic Battleeffect creation
102      * @param Battle $battle
103      * @param CombatantBehavior $hero Model record with CombatantBehavior
104      * @param CombatantBehavior $enemy Model record with CombatantBehavior
105      * @param bool $log whether the effect creation should be looged with a 
106      * Battlemessage
107      */
108     public function createEffects($battle, $hero, $enemy, $log = true) {
109         if($this->createBattleeffectID != null) {
110             $Battleeffect = clone $this->createBattleeffect;
111             
112             // See Battleeffect->initialize for further details
113             $result = $Battleeffect->call("initialize", $battle, $hero, $enemy, array(
114                 'turns' => $this->battleeffectTurns)
115             );
116             if($log) {
117                 $battleMsg = new Battlemessage("", $this);
118                 if($result == "added") {
119                     $battleMsg->msg = sprintf($this->call("getMsgResolved"), $hero->name);
120                 } elseif($result == "increasedDuration") {
121                     $battleMsg->msg = sprintf($this->call("getMsgIncreasedDuration"), $hero->name);
122                 }
123                 $battleMsg->setResult("effect", $Battleeffect);
124                 $battle->log($hero, $battleMsg);
125             }
126         }
127     }

129     /**
130      * Basic damage dealing mechanics
131      * @param Battle $battle
132      * @param CombatantBehavior $hero Model record with CombatantBehavior
133      * @param CombatantBehavior $enemy Model record with CombatantBehavior
134      * @uses getDamageFixedAmount
135      * @uses getDamageAttackStatBonus
136      * @uses getCritFactor
137      * @uses onBeforeDealingDamage
138      * @uses onAfterDealingDamage
139      * @uses BattleActionDamageEvent
140      * @uses BattleActionDamageDealtEvent
141      */
142     public function dealDamage($battle, $hero, $enemy) {
143         if($this->dealsDamage) {
144             $damageType = $this->damageType;
145 
146             // Fixed amount + variation
147             $damage = $this->call("getDamageFixedAmount");
148             // Attack stat bonus (resoluteness or willpower) + cap of said bonus
149             $damage += $this->call("getDamageAttackStatBonus", $hero);
150             /**
151              * Critical hit factor
152              * 2 for critical hits [typically], 1 for normal hits
153              */
154             $critFactor = $this->call("getCritFactor", $hero);
155             $damage *= $critFactor;
156             
157             // Bonus collection
158             $event = new BattleActionDamageEvent($battle, $hero, $enemy, $this, 
159                     $damage, $damageType);
160             // raise 2x
161             $battle->onBeforeDealDamage($event);
162             $hero->onCalcBonusDamage($event, $damageType);
163             $damage = $event->adjustStat($damage);
164             
165             // Adjust damage based on hero's attack and enemy's defense rating
166             $damage *= $this->call("getAttDefDifferenceFactor", 
167                             $hero->getAttack($this->damageAttackFactorStat) - $enemy->getDefense());
168             
169             /**
170              * Actually inflict damage
171              * This returns an integer >= 0
172              */
173             $damageDone = $enemy->takeDamage($damage, $damageType);
174 
175             $battleMsg = new Battlemessage(
176                     ($critFactor > 1
177                         ? "<span class='label label-important'>Critical Hit!</span> " : "") . 
178                     sprintf($this->call("getMsgResolved"), $hero->name), 
179                     $this);
180             $battleMsg->setResult("damage", $damageDone, $damageType);
181             $battle->log($hero, $battleMsg);
182             
183             // Notification only
184             $event = new BattleActionDamageDealtEvent($battle, 
185                     $hero, $enemy, $this, 
186                     $damageDone, $damageType
187             );
188             $battle->onAfterDealtDamage($event);
189         }
190     }

192     /**
193      * Returns damageFixedAmount + variation
194      * (based on damageFixedAmountVariation)
195      * @return float
196      */
197     public function getDamageFixedAmount() {
198         $damage = $this->damageFixedAmount;
199         // Variations
200         if(!empty($this->damageFixedAmountVariation)) {
201             $damage += mt_rand(0, $this->damageFixedAmountVariation);
202             $damage -= mt_rand(0, $this->damageFixedAmountVariation);
203         }
204         return $damage;
205     }

207     /**
208      * Returns attack stat based bonus damage
209      * @param CombatantBehavior $hero Model record with CombatantBehavior
210      * @return float
211      */
212     public function getDamageAttackStatBonus($hero) {
213         switch ($this->damageAttackFactorStat) {
214             case "resoluteness":
215             case "willpower":
216                 $attackStat = $hero->getAttack($this->damageAttackFactorStat);
217                 break;
218             default:
219                 $attackStat = 0;
220                 break;
221         }
222         $damageAttackBonus = $this->damageAttackFactor * $attackStat;
223         
224         // Limit attack stat based bonus if a limit is defined
225         if(!empty($this->damageAttackFactorCap)) {
226             $damageAttackBonus = min($damageAttackBonus, $this->damageAttackFactorCap);
227         }
228         
229         return $damageAttackBonus;
230     }

232     /**
233      * Returns the crit factor (1 if not crit, 2 if crit)
234      * @param CombatantBehavior $hero Model record with CombatantBehavior
235      * @return float
236      */
237     public function getCritFactor($hero) {
238         $critFactor = 1;
239         
240         // Get crit chance bonus
241         $event = new CollectBonusEvent($this);
242         $hero->onCalcCritChance($event);
243         $critChance = 5 + $event->getBonusPerc();
244 
245         if(mt_rand(0, 100) <= $critChance) {
246             $critFactor = 2;
247         }
248         return $critFactor;
249     }

251     /**
252      * Returns the damage modifier based on the difference of hero's
253      * attack stat and enemy's defense stat
254      * @param int $difference
255      * @return float
256      */
257     public function getAttDefDifferenceFactor($difference) {
258         // @todo load precalculated modifiers?
259         if($difference < 0) {
260             return (0.1 + 0.9 * pow(10/11, abs($difference)));
261         } else {
262             return (1 + 0.12 * pow($difference, 0.95));
263         }
264     }
