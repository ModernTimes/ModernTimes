  6 /**
  7  * Represents and resolves non-battle skills
  8  *

 16 class Skill extends BaseSkill {
 17 
 18     /**
 19      * Resolves the Skill using basic skill mechanics
 20      * @param Character $Character Character who uses the skill
 21      */
 22     public function resolveUsage($Character) {
 23         if($Character->energy < $this->costEnergy) {
 24             EUserFlash::setErrorMessage("You don't have enough energy to use that skill.");
 25         } else {
 26             $Character->decreaseEnergy($this->costEnergy);
 27             $Character->update();
 28             $this->call("createEffects", $Character);
 29         }
 30     }
 31     
 32     /**
 33      * Basic Effect creation
 34      * @param Character $Character
 35      */
 36     public function createEffects($Character) {
 37         if($this->createEffectID != null) {
 38             $CharacterEffect = $Character->getCharacterEffect($this->createEffectID);
 39             $CharacterEffect->turns += $this->effectTurns;
 40             $CharacterEffect->save();
 41         }
 42     }
 43     
 44     /**
 45      * Basic getter
 46      * @return string
 47      */
 48     public function getMsgResolved() {
 49         return $this->msgResolved;
 50     }
 51     /**
 52      * There might be a different message in case the duration of an effect
 53      * was increased (instead of a new effect getting in place)
 54      * @uses getMsgResolved
 55      * @return string
 56      */
 57     public function getMsgIncreasedDuration() {
 58         return (!empty($this->effectMsgIncreasedDuration) ? 
 59                     $this->effectMsgIncreasedDuration :
 60                     $this->getMsgResolved());
 61     }

 63     /**
 64      * Returns a string that can be used as the ocntent of a popup for this
 65      * Skill
 66      * @return string
 67      */
 68     public function getPopup() {
 69         return "<p>" . $this->desc . 
 70                ($this->costEnergy > 0 ? "<BR />&nbsp;<BR /><span class='btn btn-mini'><i class='icon-star'></i> " . $this->costEnergy . "</span>" : "") . 
 71                "</p>";
 72     }
