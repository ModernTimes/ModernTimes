  6 /**
  7  * Resolve the skill by interfering with the battle object
  8  * The following holds true for all funcitons in this class:
  9  * hero and enemy are from the Skill user's point of view
 10  * @param Battle $battle
 11  * @param Combatant $hero
 12  * @param Combatant $enemy
 13  * ToDo: source out battle action behavior (sharable with Item model)
 14  * ToDo: implement non-combat skill mechanics
 15  */
 16 
 17 class Skill extends BaseSkill {
 18 
 19     public $blocked = false;
 20 
 21     public function resolve($battle, $hero, $enemy) {
 22         if(!$this->call("checkBlocked", $battle, $hero, $enemy)) {
 23             return;
 24         }
 25         
 26         $this->call("dealDamage", $battle, $hero, $enemy);
 27         $this->call("createEffects", $battle, $hero, $enemy);
 28     }
 29     
 30     public function checkBlocked($battle, $hero, $enemy) {
 31         if($this->blocked) {
 32             $battleMsg = new Battlemessage("", $this);
 33             $battleMsg->setResult("blocked");
 34             $battle->log($hero, $battleMsg);
 35             
 36             /**
 37              * Remember that blocked is a property of the skill class and gets
 38              * carried over to future rounds unless it is reset properly
 39              */
 40             $this->blocked = false;
 41             
 42             return false;
 43         }
 44         return true;
 45     }
 46     
 47     /**
 48      * basic Battleeffect creation
 49      */
 50     public function createEffects($battle, $hero, $enemy, $log = true) {
 51         if($this->createEffect != null) {
 52             $effect = clone $this->createEffect0;
 53             
 54             // See Battleeffect->initialize for further details
 55             $result = $effect->call("initialize", $battle, $hero, $enemy, array(
 56                 'turns' => $this->effectTurns)
 57             );
 58             if($log) {
 59                 $battleMsg = new Battlemessage("", $this);
 60                 if($result == "added") {
 61                     $battleMsg->msg = sprintf($this->call("getMsgResolved"), $hero->name);
 62                 } elseif($result == "increasedDuration") {
 63                     $battleMsg->msg = sprintf($this->call("getMsgIncreasedDuration"), $hero->name);
 64                 }
 65                 $battleMsg->setResult("effect", $effect);
 66                 $battle->log($hero, $battleMsg);
 67             }
 68         }
 69     }

 71     /**
 72      * Basic damage dealing stuff
 73      */
 74     public function dealDamage($battle, $hero, $enemy) {
 75         if($this->dealsDamage) {
 76             $damage = $this->damageFixedAmount;
 77             $damageType = $this->damageType;
 78 
 79             if ($this->costEnergy == 0) {
 80                 $damage += $this->damageAttackFactor * $hero->getNormalAttack();
 81             } else {
 82                 $damage += $this->damageAttackFactor * $hero->getSpecialAttack();
 83             }
 84 
 85             // Give Battleeffects an opportunity to react
 86             $event = new CModelEvent($this, array('battle' => $battle,
 87                                                 'hero' => $hero,
 88                                                 'enemy' => $enemy,
 89                                                 'damage' => &$damage,
 90                                                 'damageType' => &$damageType));
 91             $battle->onBeforeDealingDamage($event);
 92 
 93             $damageDone = $enemy->takeDamage($damage, $damageType);
 94 
 95             $battleMsg = new Battlemessage(sprintf($this->call("getMsgResolved"), $hero->name), $this);
 96             $battleMsg->setResult("damage", $damageDone, $damageType);
 97             $battle->log($hero, $battleMsg);
 98             
 99             // Give Battleeffects an opportunity to react
100             $event = new CModelEvent($this, array('battle' => $battle,
101                                                 'hero' => $hero,
102                                                 'enemy' => $enemy,
103                                                 'damageDone' => $damageDone,
104                                                 'damageType' => $damageType));
105             $battle->onAfterDealingDamage($event);
106         }
107     }
