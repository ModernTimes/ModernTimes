 37     private function gainResource($resource, $amount, $from) {
 38         $bonusAbs = 0;
 39         $bonusPerc = 1;
 40         
 41         // Note that amount is read-only
 42         $event = new CModelEvent($this, array('bonusAbs'  => &$bonusAbs,
 43                                               'bonusPerc' => &$bonusPerc,
 44                                               'amount' => $amount,
 45                                               'from'   => $from));
 46         call_user_func(array($this, "onGaining" . ucfirst($resource)), $event);
 47         
 48         $amount = floor(
 49                     $amount * (($bonusPerc + 100) / 100) +
 50                     $bonusAbs
 51                   );
 52         
 53         call_user_func(array($this, "increase" . ucfirst($resource)), $amount);
 54     }

 97      * Adds an item to the character's inventory
 98      * @param Item $item 
 99      * @return bool, success?
100      */
101     public function gainItem($item) {
102         // d($item);
103 
104         if(!is_a($item, "Item")) {
105             // ToDo: nice exception
106             return false;
107         }
108        
109         $this->loadItems();
110 
111         $added = false;
112         foreach($this->characterItems as $characterItem) {
113             if($characterItem->item->id == $item->id) {
114                 $characterItem->n ++;
115                 $characterItem->save();
116                 $added = true;
117             }
118         }
119         if(!$added) {
120             $CharacterItem = new CharacterItems;
121             $CharacterItem->characterID = $this->id;
122             $CharacterItem->itemID = $item->id;
123             $CharacterItem->n = 1;
124             $CharacterItem->save();
125             $characterItems = $this->characterItems;
126             $characterItems[] = $CharacterItem;
127         }
128         
129         EUserFlash::setSuccessMessage("You got <b>" . $item->name . "</b>", 'gainItem id:' . $item->id);
130         return true;
131     }

220     private function increaseSubstat($stat, $amount, $generateMsg = true) {
221         // If amount is between two numbers, use RNG to determine which one to use
222         $amount = Yii::app()->tools->decideBetweenTwoNumbers($amount);
223 
224         if($amount > 0) {
225             $levelBefore = $this->getLevel();
226             $statBefore = call_user_func(array($this, "get" . ucfirst($stat) . "Base"));
227 
228             $this->{$stat . "Sub"} += $amount;
229 
230             if($generateMsg) {
231                 EUserFlash::setSuccessMessage("Your gained " . (int) $amount . " " . $stat, 'gainStat gainSubstat gain' . ucfirst($stat));
232             }
233 
234             if(call_user_func(array($this, "get" . ucfirst($stat) . "Base")) > $statBefore) {
235                 EUserFlash::setSuccessMessage("<b>Your " . $stat . " increased!</b>", 'gainStat gainNormalstat gain' . ucfirst($stat));
236             }
237             if($levelBefore < $this->getLevel()) {
238                 EUserFlash::setSuccessMessage("<b>You gained a level!</b>", 'gainStat gainLevel');
239             }
240         }

261     private function getStatBuffed($stat) {
262         $bonusAbs = 0;
263         $bonusPerc = 0;
264 
265         $event = new CEvent(null, array('bonusAbs' => &$bonusAbs,
266                                         'bonusPerc' => &$bonusPerc));
267         call_user_func(array($this, "onCalc" . ucfirst($stat)), $event);
268 
269         $ret = call_user_func(array($this, "get" . ucfirst($stat) . "Base")) *
270                     (($bonusPerc + 100) / 100) +
271                $bonusAbs;
272         //floor
273         // make sure that stats don't fall below 0 (e.g. by having bonusPerc = -120)
274         $ret = max(floor($ret), 0);
275                 
276         return $ret;
277     }
