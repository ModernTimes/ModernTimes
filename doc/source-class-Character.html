 372      * Divides xp gains among the three substats, depending on the Character
 373      * class
 374      * @uses increaseResoluteness
 375      * @uses increaseWillpower
 376      * @uses increaseCunning
 377      * @param float $xp 
 378      * @return int actual xp gain
 379      */
 380     public function increaseXp($xp) {
 381         $actualXpGain = 0;
 382         $cA = $this->getClassAttributes();
 383         $actualXpGain += $this->increaseResoluteness($xp * $cA[$this->class]['resoluteness'], false);
 384         $actualXpGain += $this->increaseWillpower($xp * $cA[$this->class]['willpower'], false);
 385         $actualXpGain += $this->increaseCunning($xp * $cA[$this->class]['cunning'], false);
 386         if($actualXpGain > 0) {
 387             EUserFlash::setSuccessMessage($actualXpGain . " experience points", 'gainStat gainXP');
 388         }
 389         return $actualXpGain;

 423     /**
 424      * Increases the indicated substat by $amount
 425      * @todo handle negative $amount
 426      * If $generateMsg: also generates an EUserFlash message to inform the user 
 427      * about this fortunate turn of events.
 428      * This is more of a setter method and does not raise any events.
 429      * @uses Tools->decideBetweenTwoNumbers
 430      * @param string $stat enum(resoluteness|cunning|willpower)
 431      * @param bool $generateMsg
 432      * @return int actual substat gains
 433      */
 434     private function increaseSubstat($stat, $amount, $generateMsg = true) {
 435         // If amount is between two numbers, use RNG to determine which one to use
 436         $amount = Yii::app()->tools->decideBetweenTwoNumbers($amount);
 437 
 438         if($amount > 0) {
 439             $levelBefore = $this->getLevel();
 440             $statBefore = call_user_func(array($this, "get" . ucfirst($stat) . "Base"));
 441 
 442             $this->{$stat . "Sub"} += $amount;
 443 
 444             if($generateMsg) {
 445                 EUserFlash::setSuccessMessage("Your gained " . (int) $amount . " " . $stat, 'gainStat gainSubstat gain' . ucfirst($stat));
 446             }
 447 
 448             if(call_user_func(array($this, "get" . ucfirst($stat) . "Base")) > $statBefore) {
 449                 EUserFlash::setSuccessMessage("<b>Your " . $stat . " increased!</b>", 'gainStat gainNormalstat gain' . ucfirst($stat));
 450             }
 451             if($levelBefore < $this->getLevel()) {
 452                 EUserFlash::setSuccessMessage("<b>You gained a level!</b>", 'gainStat gainLevel');
 453             }
 454         }
 455         return $amount;
 456     }

 571     /**
 572      * Calculates the level (x) of the Character based on mainstat (y)
 573      * y = (x−1)^2 + 4
 574      * => y-4 = (x-1)^2
 575      * => sqrt(y-4)   = x-1
 576      * => sqrt(y-4)+1 = x
 577      * @return int
 578      */
 579     public function getLevel() {
 580         $mainstatBase = $this->getMainstatBase();
 581         return floor(
 582                 sqrt(max(4,$mainstatBase) - 4)
 583                ) + 1;
 584     }
 585     /**
 586      * Returns level progression in % (0.25 for 25%)
 587      * @return float
 588      */
 589     public function getLevelProgress() {
 590         $mainstatBase = $this->getMainstatBase();
 591         $level = $this->getLevel();
 592         
 593         $statCurrentToNextLevel = pow($level, 2) - pow($level - 1, 2);
 594         $statThisLevel = $mainstatBase - pow($level - 1, 2) - 4;
 595         return $statThisLevel / $statCurrentToNextLevel;
 596     }

 636     /**
 637      * Returns how much damage the Character actually suffered
 638      * @uses onBeforeTakeDamage
 639      * @uses onAfterTakenDamage
 640      * @uses CombatantTakeDamageEvent
 641      * @uses CombatnatTakenDamageEvent
 642      * @param int $damage, how much damage the Character is to take
 643      * @param string $damageType enum(normal|vices)
 644      * @return int 
 645      */
 646     public function takeDamage($damage, $damageType) {
 647         // $debugArray["damageInitially"] = $damage;
 648         
 649         // TakeDamageEvent, collect bonuses
 650         $event = new CombatantTakeDamageEvent($this, $damage, $damageType);
 651         $this->onBeforeTakeDamage($event);
 652         $damageAdjusted = $event->adjustStat($damage);
 653         // $debugArray["damageAfterTakeDamageEvent"] = $damageAdjusted;
 654         
 655         // Reduce damage based on absolute damage reduction value
 656         $event = new CollectBonusEvent($this);
 657         $this->onCalcResistanceAbs($event);
 658         $damageAdjusted -= $event->getBonusAbs();
 659         // $debugArray["damageAfterResistanceAbs"] = $damageAdjusted;
 660         
 661         if($damageAdjusted > 0) {
 662             // Reduce damage based on resistance level
 663             $resistanceLevel = $this->getResistanceLevel($damageType);
 664             $damageAdjusted *= $this->getResistanceModifier($resistanceLevel);
 665             // $debugArray["damageAfterResistanceLevels"] = $damageAdjusted;
 666             
 667             // Beautify result
 668             $damageAdjusted = Yii::app()->tools->decideBetweenTwoNumbers(max($damageAdjusted, 0));
 669             // $debugArray['damageAfterBeautification'] = $damageAdjusted;
 670 
 671             // Actually take damage
 672             $this->decreaseHp($damageAdjusted);
 673         } else {
 674             $damageAdjusted = 0;
 675         }
 676         
 677         // takeN damage event, notification only
 678         $event = new CombatantTakenDamageEvent($this, $damageAdjusted, $damageType);
 679         $this->onAfterTakenDamage($event);
 680         
 681         // d($debugArray);
 682         return $damageAdjusted;
 683     }

 915     public function loadQuests() {
 916         if(!$this->hasRelatedQuestsFull) {
 917             foreach($this->characterQuests as $CharacterQuest) {
 918                 if($CharacterQuest->state == "ongoing") {
 919                     $CharacterQuest->quest->detachFromCharacter($this);

 924                 't.characterID=:characterID',

 933             foreach($this->characterQuests as $CharacterQuest) {

 935                         $CharacterQuest->state != "failed") {
 936 
 937                     $CharacterQuest->quest->call("initialize", $this, $CharacterQuest);
