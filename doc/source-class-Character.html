 51     /**
 52      * Gives resources to the character (or take them away)
 53      * Before it actually does, it raises a gainingResource event, to which
 54      * other code elements can react, especially Model records with
 55      * CharacterModifierBehavior.
 56      * @todo define other sources (trade, whatever)
 57      * @param string $resource enum(cash|favours|kudos)
 58      * @param float $amount
 59      * @param string $from enum(battle|encounter|quest|autosell) 
 60      * Allows event handlers to react to gainStuff events only in case the
 61      * resources come from a certain source
 62      */ 
 63     private function gainResource($resource, $amount, $from) {
 64         $bonusAbs = 0;
 65         $bonusPerc = 1;
 66         
 67         // Note that amount is read-only
 68         $event = new CModelEvent($this, array('bonusAbs'  => &$bonusAbs,
 69                                               'bonusPerc' => &$bonusPerc,
 70                                               'amount' => $amount,
 71                                               'from'   => $from));
 72         call_user_func(array($this, "onGaining" . ucfirst($resource)), $event);
 73         
 74         $amount = floor(
 75                     $amount * (($bonusPerc + 100) / 100) +
 76                     $bonusAbs
 77                   );
 78         
 79         call_user_func(array($this, "increase" . ucfirst($resource)), $amount);
 80     }

149     /**
150      * Adds an item to the character's inventory
151      * @param Item $item 
152      * @return bool success?
153      */
154     public function gainItem($item) {
155         // d($item);
156 
157         if(!is_a($item, "Item")) {
158             // ToDo: nice exception
159             return false;
160         }
161        
162         $this->loadItems();
163 
164         $added = false;
165         foreach($this->characterItems as $characterItem) {
166             if($characterItem->item->id == $item->id) {
167                 $characterItem->n ++;
168                 $characterItem->save();
169                 $added = true;
170             }
171         }
172         if(!$added) {
173             $CharacterItem = new CharacterItems;
174             $CharacterItem->characterID = $this->id;
175             $CharacterItem->itemID = $item->id;
176             $CharacterItem->n = 1;
177             $CharacterItem->save();
178             $characterItems = $this->characterItems;
179             $characterItems[] = $CharacterItem;
180         }
181         
182         EUserFlash::setSuccessMessage("You got <b>" . $item->name . "</b>", 'gainItem id:' . $item->id);
183         return true;
184     }

391     /**
392      * Increases the indicated substat by $amount
393      * @todo handle negative $amount
394      * If $generateMsg: also generates an EUserFlash message to inform the user 
395      * about this fortunate turn of events.
396      * This is more of a setter method and does not raise any events.
397      * @see Tools->decideBetweenTwoNumbers
398      * @param string $stat enum(resoluteness|cunning|willpower)
399      * @param bool $generateMsg
400      */
401     private function increaseSubstat($stat, $amount, $generateMsg = true) {
402         // If amount is between two numbers, use RNG to determine which one to use
403         $amount = Yii::app()->tools->decideBetweenTwoNumbers($amount);
404 
405         if($amount > 0) {
406             $levelBefore = $this->getLevel();
407             $statBefore = call_user_func(array($this, "get" . ucfirst($stat) . "Base"));
408 
409             $this->{$stat . "Sub"} += $amount;
410 
411             if($generateMsg) {
412                 EUserFlash::setSuccessMessage("Your gained " . (int) $amount . " " . $stat, 'gainStat gainSubstat gain' . ucfirst($stat));
413             }
414 
415             if(call_user_func(array($this, "get" . ucfirst($stat) . "Base")) > $statBefore) {
416                 EUserFlash::setSuccessMessage("<b>Your " . $stat . " increased!</b>", 'gainStat gainNormalstat gain' . ucfirst($stat));
417             }
418             if($levelBefore < $this->getLevel()) {
419                 EUserFlash::setSuccessMessage("<b>You gained a level!</b>", 'gainStat gainLevel');
420             }
421         }
422     }

454     /**
455      * Returns the Character's buffed $stat (resoluteness, cunning, willpower)
456      * Buffed = Base * BonusPerc(entage based) + BonusAbs(olute)
457      * In order to determine BonusPerc and BonusAbs, it raises a CalcStat 
458      * event, to which other code elements can react, especially Model records 
459      * with CharacterModifierBehavior.
460      * @param string $stat enum(robustness|cunning|willpower)
461      * @return int
462      */ 
463     private function getStatBuffed($stat) {
464         $bonusAbs = 0;
465         $bonusPerc = 0;
466 
467         // CEvents don't need a sender object
468         $event = new CEvent(null, array('bonusAbs' => &$bonusAbs,
469                                         'bonusPerc' => &$bonusPerc));
470         call_user_func(array($this, "onCalc" . ucfirst($stat)), $event);
471 
472         $ret = call_user_func(array($this, "get" . ucfirst($stat) . "Base")) *
473                     (($bonusPerc + 100) / 100) +
474                $bonusAbs;
475         //floor
476         // make sure that stats don't fall below 0 (e.g. by having bonusPerc = -120)
477         $ret = max(floor($ret), 0);
478                 
479         return $ret;
480     }

551     /**
552      * Calculates the level (x) of the Character based on mainstat (y)
553      * y = (x−1)^2 + 4
554      * => y-4 = (x-1)^2
555      * => sqrt(y-4)   = x-1
556      * => sqrt(y-4)+1 = x
557      * @return int
558      */
559     public function getLevel() {
560         $mainstatBase = $this->getMainstatBase();
561         return floor(
562                 sqrt(max(4,$mainstatBase) - 4)
563                ) + 1;
564     }
565     /**
566      * Returns level progression in % (0.25 for 25%)
567      * @return float
568      */
569     public function getLevelProgress() {
570         $mainstatBase = $this->getMainstatBase();
571         $level = $this->getLevel();
572         
573         $statCurrentToNextLevel = pow($level, 2) - pow($level - 1, 2);
574         $statThisLevel = $mainstatBase - pow($level - 1, 2) - 4;
575         return $statThisLevel / $statCurrentToNextLevel;
576     }

689     /**
690      * Adds a CharacterEffects record to the Character record
691      * @see CharacterEffects
692      * @param CharacterEffects $characterEffect

694     public function addEffect($characterEffect) {
695         /**
696          * Remember: $this->characterEffects is NOT a real property, but the 
697          * result of a function call!
698          */
699         $charEffects = $this->characterEffects;
