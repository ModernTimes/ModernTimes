  3 /**
  4  * Mammon waits until the Character exploits their contacts for cash a
  5  * number of times.
  6  * 
  7  * This Quest is added for new Characters (state = unavailable).
  8  * 
  9  * Upon completion, the Character can summon Mammon.
 10  * 
 11  * @uses SpecialnessBehavior
 12  * @uses CharacterQuests
 13  * @package Quests
 14  */
 15 
 16 class Greed1Quest extends CBehavior {
 17 
 18     /**
 19      * How often the Character has to exploit a contact for cash
 20      * @const int 
 21      */
 22     const numberOfFavors = 20;
 23     
 24     /**
 25      * ID of the "Ask for cash" favor
 26      * @const int 
 27      */
 28     const favorID = 2;
 29     
 30     /**
 31      * Encounter record id of Mommon saying hello the first time
 32      * @const int 
 33      */
 34     const mommonCallsEncounterID = 2;
 35 
 36     /**
 37      * Set the initial parameters for the Quest.
 38      * In this case: a counter
 39      */
 40     public function setInitialParams() {
 41             $this->owner->params['n'] = 0;
 42             $this->owner->saveParams();
 43     }
 44 
 45     /**
 46      * Attaches custom event handlers to a Character
 47      * @param Character $Character 
 48      * @return void
 49      */
 50     public function attachToCharacter($Character) {
 51         $Character->onExploit = array($this, 'reactToOnExploit');
 52     }
 53 
 54     /**
 55      * Detaches custom event handlers from a Character
 56      * @param Character $Character 
 57      */
 58     public function detachFromCharacter($Character) {
 59         $Character->detachEventHandler("onExploit", array($this, 'reactToOnExploit'));
 60     }
 61 
 62     /**
 63      * Checks if the Character exploited for cash, and if so, increases n by 1
 64      * @param ExploitEvent $event 
 65      * @return void
 66      */
 67     public function reactToOnExploit($event) { 
 68         $Favor = $event->getFavor();
 69         if($Favor->id == self::favorID) {
 70             $this->owner->params['n'] ++;
 71             if($this->owner->params['n'] >= self::numberOfFavors) {
 72                 $this->owner->setState("succeeded");
 73             } else {
 74                 $this->owner->saveParams();
 75             }
 76         }
 77     }
 78     
 79     /**
 80      * Adds the "Mommon says hello" encounter to the encounter queue
 81      * @param QuestChangeStateEvent $event 
 82      */
 83     public function reactToOnSucceeded($event) {
 84         $CharacterEncounter = new CharacterEncounters();
 85         $CharacterEncounter->characterID = $this->owner->CharacterQuest->characterID;
 86         $CharacterEncounter->encounterID = self::mommonCallsEncounterID;
 87         $CharacterEncounter->delay = 0;
 88         $CharacterEncounter->save();
 89     }
 90     
 91     /**
 92      * Sets the CharacterQuest to visible
 93      * @param QuestChangeStateEvent $event 
 94      */
 95     public function reactToOnCompleted($event) {
 96         $this->owner->visible = 1;
 97         // Reset params and update CharacterQuest record
 98         $this->owner->reactToOnCompleted($event);
 99     }
100 }
