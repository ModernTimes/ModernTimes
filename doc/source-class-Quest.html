 42     /**
 43      * Initializes the quest:
 44      * - Hooks into Character's events
 45      * - Sets links to a Character and the CharacterQuests record
 46      * - Loads its params (or initializes them)
 47      * - Hooks its own reactToOnX to its onX
 48      * Is only called if CharacterQuest->state is not completed or failed
 49      * @uses loadParams
 50      * @uses setInitialParams
 51      * @uses onChangeState
 52      * @uses reactToOnChangeState
 53      * @uses callReactToOnChangeState
 54      * @param Character $Character
 55      * @param CharacterQuests $CharacterQuest 
 56      */
 57     public function initialize($Character, $CharacterQuest) {
 58         $this->Character = $Character;
 59         $this->CharacterQuest = $CharacterQuest;
 60         
 61         $this->call('attachToCharacter', $Character);
 62         
 63         $this->loadParams();
 64         if(empty($this->params)) {
 65             $this->call('setInitialParams');
 66         }
 67         
 68         $this->onChangeState = array($this, 'callReactToOnChangeState');
 69         $this->onUnavailable = array($this, 'callReactToOnUnavailable');
 70         $this->onAvailable = array($this, 'callReactToOnAvailable');
 71         $this->onOngoing = array($this, 'callReactToOnOngoing');
 72         $this->onCompleted = array($this, 'callReactToOnCompleted');
 73         $this->onRejected = array($this, 'callReactToOnRejected');
 74         $this->onFailed = array($this, 'callReactToOnFailed');
 75     }

102     /**
103      * - Sets the current state of the quest
104      * - Updates $this->CharacterQuest if indicated
105      * - Raises an onChangeState event
106      * @uses CharacterQuests
107      * @uses QuestChangeStateEvent
108      * @uses onChangeStat
109      * @param string $state enum(unavailable|available|ongoing|completed|rejected|failed)
110      * @param bool $update default true
111      */
112     public function setState($state, $update = true) {
113         $allowedStates = array("unavailable", "available", "ongoing",
114                                "completed", "rejected", "failed");
115         if(in_array($state, $allowedStates) && 
116                 $state != $this->CharacterQuest->state) {
117             
118             $this->CharacterQuest->state = $state;
119             if($update) {
120                 $this->CharacterQuest->update();
121             }
122 
123             $event = new QuestChangeStateEvent(
124                     $this, 
125                     $this->CharacterQuest->state,
126                     $state);
127             $this->onChangeState($event);
128         }
129     }

131     /**
132      * Saves the serialized param attribute to CharacterQuest record
133      * @uses CharacterQuests
134      * @param bool $update default true
135      */
136     public function saveParams($update = true) {
137         if(!empty($this->params)) {
138             $this->CharacterQuest->params = serialize($this->params);
139         } else {
140             $this->CharacterQuest->params = null;
141         }
142         if($update) {
143             $this->CharacterQuest->update();
144         }
145     }

224     /**
225      * Event handler
226      * Raises a more specific event depending on the new state
227      * @param QuestChangeStateEvent $event 
228      */
229     public function reactToOnChangeState($event) { 
230         switch($event->stateAfter) {
231             case "unavailable":
232                 $this->onUnavailable($event);
233                 break;
234             case "available":
235                 $this->onAvailable($event);
236                 break;
237             case "ongoing":
238                 $this->onOngoing($event);
239                 break;
240             case "completed":
241                 $this->onCompleted($event);
242                 break;
243             case "failed":
244                 $this->onFailed($event);
245                 break;
246             case "rejected":
247                 $this->onRejected($event);
248                 break;
249         }
250     }
