 3 /**
 4  * hero is the babble skill user
 5  * 
 6  * Mechanics:
 7  * - all skills of subtype babbling add BabbleComboEffect with charges = 1 or
 8  *   increase charges by 1 if BabblecomboEffect is already in place
 9  * - skills of subtype babbling do more damage: *= pow(base, charges)
10  * - If hero uses a skill of subtype != babbling, BabblecomboEffect 
11  *   is lost
12  */
13 
14 class BabbleComboEffect extends CBehavior {
15 
16     const BabbleComboEffect_base = 1.1;
17     
18     public function attachToBattle($battle) {
19         $this->owner->attachToBattle($battle);
20         $battle->onAfterAction = array($this, 'reactToOnAfterAction');
21         $battle->onBeforeDealingDamage = array($this, 'reactToOnBeforeDealingDamage');
22     }
23 
24     // Shows the damage increasing effect for babbling akills in %.
25     public function getPopup() {
26         return $this->owner->desc . "<BR />&nbsp;<BR /><b>" . 
27                floor(pow(self::BabbleComboEffect_base, $this->owner->charges) * 100 - 100) . 
28                "% more obnoxious already!</b>";
29     }
30     
31     /**
32      * Damage of skills with subtype babbling increases exponentially,
33      * depending on charges
34      */
35     public function reactToOnBeforeDealingDamage($event) {
36         if($this->owner->active &&
37            $event->params['battle']->getCombatantString($event->params['hero']) == $this->owner->heroString &&
38            $event->sender->owner->subType == "babbling") {
39             
40             $event->params['damage'] = $event->params['damage'] * 
41                                        pow(self::BabbleComboEffect_base, $this->owner->charges);
42         }
43     }
44 
45     /**
46      * ... until the ComboEffect is interrupted by a non-babbling skill
47      * the function also increases charges by 1 in case hero uses another skill of subtype babbling
48      */
49     public function reactToOnAfterAction($event) {
50         if($event->sender->getCombatantString($event->params['hero']) == $this->owner->heroString) {
51             if ($event->params['action']->actionType == "personal" &&
52                 $event->params['action']->subType != "babbling") {
53             
54                     $this->owner->active = false;
55 
56                     $battleMsg = new Battlemessage($event->params['hero'] . " loses " . Yii::app()->tools->getPossessivePronoun($event->params['hero']) . " babble momentum");
57                     $event->sender->log($event->params['hero'], $battleMsg);
58             }
59             
60             // Increase charges if hero uses another babbling skill
61             if ($event->params['action']->actionType == "personal" &&
62                 $event->params['action']->subType == "babbling") {
63                 
64                 $this->owner->charges++;
65 
66                 $battleMsg = new Battlemessage("", $event->params['action']);
67                 if($this->owner->charges == 1) {
68                     $battleMsg->msg = $event->params['hero']->name . " builds up some babble momentum";
69                     $event->sender->log($event->params['hero'], $battleMsg);
70                 } else {
71                     // $battleMsg->msg = $event->params['hero']->name . " increases " . Yii::app()->tools->getPossessivePronoun($event->params['hero']) . " babble momentum";
72                     // $event->sender->log($event->params['hero'], $battleMsg);
73                 }
74             }
75         }
76     }
77 }
