  5 /**
  6  * data basis for Charactermodifier behavior for items, skills, etc.
  7  * adds a couple of standard event listeners to stat calculation events in 
  8  * the Character model
  9  * Can be "overridden" by specialness behavior classes for items, skills, etc.
 10  */
 11 
 12 class Charactermodifier extends BaseCharactermodifier {
 13 
 14     /**
 15      * implemented so far:
 16      * - dropCash/Favours/Kudos Abs/Perc
 17      * - dropItem Perc
 18      * - gainXp/Resoluteness/Willpower/Cunning Abs/Perc
 19      *
 20      * If you add more event handlers, don't forget to add a line to
 21      * detachFromCharacter()
 22      *
 23      * "$component->onXXX = y" is the same as "$component->onXXX->add(y)"
 24      * 
 25      * @param Character $Character
 26      */
 27     public function attachToCharacter($Character) {
 28         if($this->hp != 0 || $this->hpPerc != 0) {
 29             $Character->onCalcHp = array($this, 'reactToOnCalcHp');
 30         }
 31         if($this->energy != 0 || $this->energyPerc != 0) {
 32             $Character->onCalcEnergy = array($this, 'reactToOnCalcEnergy');
 33         }
 34         if($this->resoluteness != 0 || $this->resolutenessPerc != 0) {
 35             $Character->onCalcResoluteness = array($this, 'reactToOnCalcResoluteness');
 36         }
 37         if($this->willpower != 0 || $this->willpowerPerc != 0) {
 38             $Character->onCalcWillpower = array($this, 'reactToOnCalcWillpower');
 39         }
 40         if($this->cunning != 0 || $this->cunningPerc != 0) {
 41             $Character->onCalcCunning = array($this, 'reactToOnCalcCunning');
 42         }
 43         if($this->dropCash != 0 || $this->dropCashPerc != 0) {
 44             $Character->onGainingCash = array($this, 'reactToOnGainingCash');
 45         }
 46         if($this->dropFavours != 0 || $this->dropFavoursPerc != 0) {
 47             $Character->onGainingFavours = array($this, 'reactToOnGainingFavours');
 48         }
 49         if($this->dropKudos != 0 || $this->dropKudosPerc != 0) {
 50             $Character->onGainingKudos = array($this, 'reactToOnGainingKudos');
 51         }
 52         if($this->dropItemPerc != 0) {
 53             $Character->onCalcDropItemBonus = array($this, 'reactToOnCalcDropItemBonus');
 54         }
 55     }

 57     /**
 58      * @param Character $Character
 59      */
 60     public function detachFromCharacter($Character) {
 61         $Character->detachEventHandler("onCalcHp", array($this, 'reactToOnCalcHp'));
 62         $Character->detachEventHandler("onCalcEnergy", array($this, 'reactToOnCalcEnergy'));
 63 
 64         $Character->detachEventHandler("onCalcResoluteness", array($this, 'reactToOnCalcResoluteness'));
 65         $Character->detachEventHandler("onCalcWillpower", array($this, 'reactToOnCalcWillpower'));
 66         $Character->detachEventHandler("onCalcCunning", array($this, 'reactToOnCalcCunning'));
 67 
 68         $Character->detachEventHandler("onGainingCash", array($this, 'reactToOnGainingCash'));
 69         $Character->detachEventHandler("onGainingFavours", array($this, 'reactToOnGainingFavours'));
 70         $Character->detachEventHandler("onGainingKudos", array($this, 'reactToOnGainingKudos'));
 71         
 72         $Character->detachEventHandler("onCalcDropItemBonus", array($this, 'reactToOnCalcDropItemBonus'));
 73     }

 77         $event->params['bonusPerc'] += $this->hpPerc;

 96     /**
 97      * standard behavior is to react to gains from battles only
 98      * Can be "overridden" by special behavior classes
 99      * $event->params: float amount (read only)
100      *                 enum(battle, encounter, ...) from (read only)
101      *                 float bonusAbs
102      *                 float bonusPerc
103      * @param array $acceptFrom, list of scenarios in which the event observer
104      * should do its work
105      */
106     public function reactToOnGainingCash($event, $acceptFrom = array('battle')) {
107         if(in_array($event->params['from'], $acceptFrom)) {
108             $event->params['bonusAbs'] += $this->dropCash;
109             $event->params['bonusPerc'] += $this->dropCashPerc;
110         }
111     }
112     public function reactToOnGainingFavours($event, $acceptFrom = array('battle')) {
113         if(in_array($event->params['from'], $acceptFrom)) {
114             $event->params['bonusAbs'] += $this->dropFavours;
115             $event->params['bonusPerc'] += $this->dropFavoursPerc;
116         }
117     }
118     public function reactToOnGainingKudos($event, $acceptFrom = array('battle')) {
119         if(in_array($event->params['from'], $acceptFrom)) {
120             $event->params['bonusAbs'] += $this->dropKudos;
121             $event->params['bonusPerc'] += $this->dropKudosPerc;
122         }
123     }
124     
125     public function reactToOnCalcDropItemBonus($event) {
126         $event->params['bonusPerc'] += $this->dropItemPerc;
127     }
