 54     /**
 55      * Returns how much damage the Monster actually suffered
 56      * @uses onBeforeTakeDamage
 57      * @uses onAfterTakenDamage
 58      * @uses CombatantTakeDamageEvent
 59      * @uses CombatnatTakenDamageEvent
 60      * @param int $damage, how much damage the Monster is to take
 61      * @param string $damageType enum(normal|vices)
 62      * @return int 
 63      */
 64     public function takeDamage($damage, $damageType) {
 65         // TakeDamageEvent, collect bonuses
 66         $event = new CombatantTakeDamageEvent($this, $damage, $damageType);
 67         $this->onBeforeTakeDamage($event);
 68         
 69         $damageAdjusted = floor($event->adjustStat($damage));
 70         $damageAdjusted = max($damageAdjusted, 0);
 71         
 72         $this->decreaseHp($damageAdjusted);
 73 
 74         // takeN damage event, notification only
 75         $event = new CombatantTakenDamageEvent($this, $damageAdjusted, $damageType);
 76         $this->onAfterTakenDamage($event);
 77         
 78         return $damageAdjusted;
 79     }

 81     /**
 82      * Decides for each potential item drop whether it is indeed dropped or not
 83      * dropItemPerc are percentage point increasers/decreasers as usual
 84      * @param int $dropItemPerc bonus to the drop rate
 85      * @return array of Item records
 86      */
 87     public function dropItems($dropItemPerc = 0) {
 88         $loot = array();
 89         foreach($this->monsterItems as $monsterItem) {
 90             $prob = max(0, min(1, $monsterItem->prob * (($dropItemPerc + 100) / 100)));
 91             $rand = mt_rand(0, 1000000);
 92             if($rand <= $prob * 1000000) {
 93                 $loot[] = $monsterItem->item;
 94             }
 95         }
 96         return $loot;
 97     }
