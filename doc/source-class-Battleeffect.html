 64     /**
 65      * Initializes the Battleeffect
 66      * Attaches itself to the Battle, or increases the duration of an
 67      * existing (same) Battleeffect 
 68      * @param Battle $battle
 69      * @param CombatantBehavior $hero Model record with CombatantBehavior
 70      * @param CombatantBehavior $enemy Model record with CombatantBehavior
 71      * @param array $options 
 72      * @return string enum(added|increasedDuration|notAdded)
 73      */
 74     public function initialize($battle, $hero, $enemy, $options = array()) {
 75         $options = array_merge(
 76             // The default options
 77             array(
 78                 'autoAttach' => true,
 79                 'turns' => $this->turns,
 80             ),
 81             // The specified options
 82             $options
 83         );        
 84 
 85         /**
 86          * up to this point: hero = combatant who brought this effect into play
 87          * adjust so that heroString is the owner of the buff/debuff
 88          */
 89         if($this->buff) {
 90             $this->heroString = $battle->getCombatantString($hero);
 91             $this->enemyString = $battle->getCombatantString($enemy);
 92         } else {
 93             $this->heroString = $battle->getCombatantString($enemy);
 94             $this->enemyString = $battle->getCombatantString($hero);
 95         }
 96         
 97         $this->active = true;
 98         $this->turns = $options['turns'];
 99 
100         if($this->blocks && $this->blockNumberOfBlocks > 0) {
101             $this->charges = $this->blockNumberOfBlocks;
102         }
103         
104         /**
105          * Adds the effect in case it's not already active
106          * Unless the effect should only be active once per combatant,
107          * in which case the duration will be increased
108          */
109         if($options['autoAttach']) {
110             if(!$this->singleton || !$battle->battleeffects->contains($this)) {
111                 $battle->addEffect($this);
112                 return "added";
113             } elseif ($this->increaseDuration) {
114                 $battle->increaseEffectDuration($this);
115                 return "increasedDuration";
116             }
117         }
118         return "notAdded";
119     }

197     /**
198      * Decreases the duration of the Battleeffect. If it goes down to 0, it
199      * deactivates itself, adds a deactivation Battlemessage and waits for
200      * Battle->__sleep to be destroyed
201      * "Override" and extend by SpecialnessBehavior classes as necessary
202      * @param CEvent $event 
203      */
204     public function reactToOnAfterRound($event) {
205         $this->turns --;
206         if($this->turns == 0) {
207             $this->active = 0;
208 
209             $battleMsg = new Battlemessage(sprintf($this->msgExpire, $event->sender->{$this->heroString}->name));
210             $event->sender->log($event->sender->{$this->heroString}, $battleMsg);
211         }
212     }

214     /**
215      * Checks if a battle action is blocked by this Battleeffect
216      * "Override" and extend by SpecialnessBehavior classes as necessary
217      * @todo check typeOfActions
218      * @param CEvent $event 
219      */
220     public function reactToOnBeforeAction($event) {
221         if($this->blocks &&
222            $this->active &&
223            $event->sender->getCombatantString($event->params['hero']) == $this->heroString &&
224            !$event->action->blocked) {
225             
226             if($this->blockChance != 1) {
227                 $rand = mt_rand(0,100);
228                 if($rand > $this->blockChance * 100) {
229                     return;
230                 }
231             }
232             
233             if($event->action->call("setBlocked")) {
234                 $this->blockNumberOfBlocks --;
235                 $this->charges --;
236                 
237                 $battleMsg = new Battlemessage(sprintf($this->msgBlock, $event->hero->name, $event->action->name));
238                 $event->sender->log($event->hero, $battleMsg);
239                         
240                 if($this->blockNumberOfBlocks == 0) {
241                     $this->active = false;
242 
243                     $battleMsg = new Battlemessage(sprintf($this->msgExpire, $event->hero->name));
244                     $event->sender->log($event->hero, $battleMsg);
245                 }
246             }
247         }
248     }
