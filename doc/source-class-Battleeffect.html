 55      *
 56      * @param Battle $battle
 57      * @param Combatant $hero
 58      * @param Combatant $enemy
 59      * @param array|mixed $options
 60      * @return enum(added, increasedDuration, notAdded)
 61      */
 62     public function initialize($battle, $hero, $enemy, $options = array()) {
 63         $options = array_merge(
 64             // The default options
 65             array(
 66                 'autoAttach' => true,
 67                 'turns' => $this->turns,
 68             ),
 69             // The specified options
 70             $options
 71         );        
 72 
 73         /**
 74          * up to this point: hero = combatant who brought this effect into play
 75          * adjust so that heroString is the owner of the buff/debuff
 76          */
 77         if($this->buff) {
 78             $this->heroString = $battle->getCombatantString($hero);
 79             $this->enemyString = $battle->getCombatantString($enemy);
 80         } else {
 81             $this->heroString = $battle->getCombatantString($enemy);
 82             $this->enemyString = $battle->getCombatantString($hero);
 83         }
 84         
 85         $this->active = true;
 86         $this->turns = $options['turns'];
 87 
 88         if($this->blocks && $this->blockNumberOfBlocks > 0) {
 89             $this->charges = $this->blockNumberOfBlocks;
 90         }
 91         
 92         /**
 93          * Adds the effect in case it's not already active
 94          * Unless the effect should only be active once per combatant,
 95          * in which case the duration will be increased
 96          */
 97         if($options['autoAttach']) {
 98             if(!$this->singleton || !$battle->battleeffects->contains($this)) {
 99                 $battle->addEffect($this);
100                 return "added";
101             } elseif ($this->increaseDuration) {
102                 $battle->increaseEffectDuration($this);
103                 return "increasedDuration";
104             }
105         }
106         return "notAdded";
107     }

167     public function reactToOnBeforeAction($event) {
168         if($this->blocks &&
169            $this->active &&
170            $event->sender->getCombatantString($event->params['hero']) == $this->heroString &&
171            !$event->params['action']->blocked) {
172             
173             if($this->blockChance != 1) {
174                 $rand = mt_rand(0,100);
175                 if($rand > $this->blockChance * 100) {
176                     return;
177                 }
178             }
179             
180             if($event->params['action']->call("setBlocked")) {
181                 $this->blockNumberOfBlocks --;
182                 $this->charges --;
183                 
184                 $battleMsg = new Battlemessage(sprintf($this->msgBlock, $event->params['hero']->name, $event->params['action']->name));
185                 $event->sender->log($event->params['hero'], $battleMsg);
186                         
187                 if($this->blockNumberOfBlocks == 0) {
188                     $this->active = false;
189 
190                     $battleMsg = new Battlemessage(sprintf($this->msgExpire, $event->params['hero']->name));
191                     $event->sender->log($event->params['hero'], $battleMsg);
192                 }
193             }
194         }
