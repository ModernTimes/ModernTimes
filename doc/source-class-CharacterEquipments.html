  5 /**
  6  * Basic HAS_MANY association model
  7  * Which equipment configurations has a character defined?
  8  * Only one can be active at a given time
  9  * 
 10  * See BaseCharacterEquipments for a list of attributes and related Models
 11  * 
 12  * Eager loads weapon, offhand, accessoryA, B, and C. Use resetScope() to
 13  * prevent that.
 14  * 
 15  * @see Character
 16  * @see Item
 17  * @package Character.Relations
 18  */
 19

 22     /**
 23      * Attaches all Item model callbacks to the character event handlers
 24      * Care: The giix Model generator adds a "0" after the item slot names, 
 25      *       for whatever reason
 26      * @param Character $Character 
 27      */
 28     public function attachToCharacter($Character) {
 29         if(is_a($this->weapon0, "Item")) {
 30             $this->weapon0->call("attachToCharacter", $Character);
 31         }
 32         if(is_a($this->offhand0, "Item")) {
 33             $this->offhand0->call("attachToCharacter", $Character);
 34         }
 35         if(is_a($this->accessoryA0, "Item")) {
 36             $this->accessoryA0->call("attachToCharacter", $Character);
 37         }
 38         if(is_a($this->accessoryB0, "Item")) {
 39             $this->accessoryB0->call("attachToCharacter", $Character);
 40         }
 41         if(is_a($this->accessoryC0, "Item")) {
 42             $this->accessoryC0->call("attachToCharacter", $Character);
 43         }
 44     }

 46     /**
 47      * Detaches all Item model callbacks from the character event handlers
