 10      * returns an integer number adjacent to $number
 11      * 3.38 has a 38% chance to return 4 and a 62% chance to return 3.
 12      * @param float $number
 13      * @return int
 14      */
 15     function decideBetweenTwoNumbers($number) {
 16         $modulo = $number - floor($number);
 17         $number = floor($number);
 18         if($modulo > 0) {
 19             $rand = mt_rand(0, 100);
 20             if($rand <= $modulo * 100) {
 21                 $number ++;
 22             }
 23         }
 24         return $number;
 25     }

 46     /**
 47      * Adds an effect to the active character
 48      * @param Effect|int|string $effect, Effect model or its PK or its name
 49      * @param int $turns, number of turns for which the effect is to be active
 50      *                    0 means that it is only used during the current battle
 51      * options
 52      * @param bool 'addTurns', whether to increase the number of turns when the 
 53      *                         effect is already in place instead of adding a 
 54      *                         second effect of the same kind. Default is true.
 55      * @param int 'characterID', PK of the character that the effect is to be
 56      *                           attached to
 57      * 
 58      * @return boolean, whether or not the effect was added
 59      * ToDo: Make it possible to add effects to other characters
 60      */
 61     function addEffect($effect, $turns = 0, $options = array()) {
 62         $options = array_merge(
 63             // The default options
 64             array(
 65                 'addTurns' => true,
 66                 'characterID' => CD()->id,
 67             ),
 68             // The specified options
 69             $options
 70         );
 71         
 72         // d($effect);
 73 
 74         if ($turns == 0) {
 75             Yii::trace("Cannot add effect with duration of 0 turns");
 76             return false;
 77         }
 78         
 79         if(is_int($effect)) {
 80             $effectModel = Effect::model()->findByPk($effect);
 81         } elseif (is_string($effect)) {
 82             $effectModel = Effect::model()->find("name = '" . $effect . "'");
 83         } elseif (is_a($effect, "Effect")) {
 84             $effectModel = $effect;
 85         }
 86         if(!is_a($effectModel, "Effect")) {
 87             return false;
 88         }
 89         
 90         $character = CD();
 91         
 92         if ($character->hasEffect($effectModel)) {
 93             if($options['addTurns']) {
 94                 // returns CharacterEffects model, not Effect model
 95                 $effectInPlace = $character->getEffect($effectModel);
 96                 $effectInPlace->increaseDuration($turns);
 97                 if ($effectInPlace->save()) {
 98                     Yii::trace("Effect already in place, increased turns by " . $turns);
 99                     return true;
100                 } else {
101                     return false;
102                 }
103             } else {
104                 Yii::trace("Effect already in place, nothing to do here");
105             }
106         } else {
107             $characterEffect = new CharacterEffects();
108             $characterEffect->characterID = $character->id;
109             $characterEffect->effectID = $effectModel->id;
110             $characterEffect->effect = $effectModel;
111             $characterEffect->turns = $turns;
112             
113             if ($characterEffect->save()) {
114                 $character->addEffect($characterEffect);
115                 Yii::trace("Added effect '" . $effectModel->name . "' for " . $turns . " turns");
116                 return true;
117             } else {
118                 return false;
119             }
120         }
121     }
