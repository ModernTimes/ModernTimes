 35     /**
 36      * returns an integer number adjacent to $number
 37      * 3.38 has a 38% chance to return 4 and a 62% chance to return 3.
 38      * @param float $number
 39      * @return int
 40      */
 41     function decideBetweenTwoNumbers($number) {
 42         $modulo = $number - floor($number);
 43         $number = floor($number);
 44         if($modulo > 0) {
 45             $rand = mt_rand(0, 100);
 46             if($rand <= $modulo * 100) {
 47                 $number ++;
 48             }
 49         }
 50         return $number;
 51     }

124     /**
125      * Adds an effect to a character
126      * @param Character $Character Character record that the effect is to be
127      * added to
128      * @param mixed $effect Effect|int|string Effect model or its PK or its name
129      * @param int $turns number of turns for which the effect is to be active
130      *                   0 means that it is only used during the current battle
131      * @param array $options
132      * param bool 'addTurns', whether to increase the number of turns when the 
133      *                         effect is already in place instead of adding a 
134      *                         second effect of the same kind. Default is true.
135      * 
136      * @return boolean whether or not the effect was added
137      */
138     function addEffect($Character, $Effect, $turns = 0, $options = array()) {
139         $options = array_merge(
140             // The default options
141             array(
142                 'addTurns' => true
143             ),
144             // The specified options
145             $options
146         );
147         
148         // d($effect);
149 
150         if ($turns == 0) {
151             Yii::trace("Cannot add effect with duration of 0 turns");
152             return false;
153         }
154         
155         if(is_int($Effect)) {
156             $Effect = Effect::model()->findByPk($Effect);
157         } elseif (is_string($Effect)) {
158             $Effect = Effect::model()->find("name = '" . $Effect . "'");
159         }
160         if(!is_a($Effect, "Effect")) {
161             return false;
162         }
163         
164         if ($Character->hasEffect($Effect)) {
165             if($options['addTurns']) {
166                 // returns CharacterEffects model, not Effect model
167                 $effectInPlace = $Character->getCharacterEffect($Effect);
168                 $effectInPlace->increaseDuration($turns);
169                 if ($effectInPlace->save()) {
170                     Yii::trace("Effect already in place, increased turns by " . $turns);
171                     return true;
172                 } else {
173                     return false;
174                 }
175             } else {
176                 Yii::trace("Effect already in place, nothing to do here");
177             }
178         } else {
179             $CharacterEffect = new CharacterEffects();
180             $CharacterEffect->characterID = $Character->id;
181             $CharacterEffect->effectID = $Effect->id;
182             $CharacterEffect->effect = $Effect;
183             $CharacterEffect->turns = $turns;
184             
185             if ($CharacterEffect->save()) {
186                 $Character->addCharacterEffect($CharacterEffect);
187                 Yii::trace("Added effect '" . $Effect->name . "' for " . $turns . " turns");
188                 return true;
189             } else {
190                 return false;
191             }
192         }
193     }
