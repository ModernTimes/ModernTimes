114     /**
115      * Stops the battle and handle consequences of winning + loosing
116      * Also does some cleaning up
117      */
118     public function stop() {
119         if($this->combatantA->hp > 0) {
120             $this->winnerType = "player";
121             $this->winnerID = $this->combatantAID;
122         } elseif ($this->combatantB->hp > 0) {
123             $this->winnerType = ($this->type == "pvp" ? "player" : "monster");
124             $this->winnerID = $this->combatantBID;
125         } else {
126             $this->winnerType = "draw";
127         }
128 
129         // Bounty + XP
130         if($this->type == "monster" && $this->combatantB->hp <= 0) {
131             $loot = $this->combatantB->dropItems($this->combatantA->getDropItemPerc());
132             $this->combatantA->gainItems($loot);
133 
134             $this->combatantA->gainCash($this->combatantB->dropCash, "battle");
135             $this->combatantA->gainFavours($this->combatantB->dropFavours, "battle");
136             $this->combatantA->gainKudos($this->combatantB->dropKudos, "battle");
137 
138             /**
139              * monster->xp can be defined to deviate from the standard
140              * attack / 2 xp reward
141              */
142             if(isset($this->combatantB->xp)) {
143                 $this->combatantA->gainXp($this->combatantB->xp, "battle");
144             } else {
145                 $this->combatantA->gainXp($this->combatantB->attack / 4, "battle");
146             }
147         }
148 
149         $this->combatantA->ongoingBattleID = null;
150         if($this->type == "pvp") {
151             // @todo Message to PVP enemy
152             $this->combatantB->ongoingBattleID = null;
153         }
154 
155         $this->state = "resolved";
156         $this->objectState = "";
157 
158         /**
159          * ToDo: why does backupbehavior think that no attribute changed
160          *       if we set ongoingBattleID to null?
161          */
162         // $this->combatantA->save();
163         $this->combatantA->update();
164         if($this->type == "pvp") {
165             $this->combatantB->save();
166         }
167 
168         $this->update();
169         $this->reconstructCombatants();
170         
171         // Notify the world about this result
172         $event = new BattleEvent($this);
173         $this->combatantA->onFinishedBattle($event);
174         if($this->type == "pvp") {
175             $this->combatantB->onFinishedBattle($event);
176         }
177     }

179     /**
180      * Checks if all combatants have valid actions defined. Calls
181      * calculateRound if that is so.
182      * Validity of $playerAction is checked by controller or is provided by
183      * a Monster model directly
184      * @uses calculateRound
185      * @param mixed Battleskill or Item record
186      */
187     public function playerAction($playerAction = null) {
188         // Yii::trace("playerAction. Action: " . $playerAction->name);
189         
190         // Did the player define a new and valid action?
191         if($playerAction != $this->{$this->getHeroString() . "Action"} &&
192            is_object($playerAction)) {
193 
194             $this->{$this->getHeroString() . "Action"} = $playerAction;
195 
196             // Let the monster act
197             if($this->type == "monster") {
198                 $this->combatantBAction = $this->combatantB->call("act", $this);
199             }
200 
201             // If enemy has already acted: continue!
202             if(is_object($this->{$this->getEnemyString() . "Action"})) {
203 
204                     $this->calculateRound();
205             } else {
206                 // ToDo: Print waiting for opponent message
207             }
208         } else {
209             Yii::trace("Player action is the same as before (or invalid)");
210         }

213     /**
214      * - resolve block actions
215      * - resolve delayed effects
216      * - resolve defensive actions
217      * - resolve offensive actions
218      * - create new battleeffects
219      * 
220      * @uses onBeforeRound with BattleEvent
221      * @uses onAfterRound with BattleEvent
222      * @uses onBeforeAction with BattleActionEvent
223      * @uses onAfterAction with BattleActionEvent
224      * @uses BattleEvent
225      * @uses BattleActionEvent
226      */
227     public function calculateRound() {
228         Yii::trace("calculateRound");
229 
230         $this->nextRound();
231         $this->onBeforeRound(new BattleEvent($this));
232 
233         /**
234          * Resolve actions
235          * Action calls expects $battle, $hero, $enemy, from the action's user 
236          * point of view
237          * If actions happen in the same battle phase, RNG decides who acts 
238          * first. Otherwise, actions resolve in the order of the battle phases
239          */
240         if($this->combatantAAction->battlePhase == $this->combatantBAction->battlePhase) {
241             $rand = mt_rand(0,100);
242             if($rand >= 50) {
243                 $first = "combatantA";
244             } else {
245                 $first = "combatantB";
246             }
247         } elseif ($this->combatantAAction->battlePhase == "block") {
248             $first = "combatantA";
249         } elseif ($this->combatantBAction->battlePhase == "block") {
250             $first = "combatantB";
251         } elseif ($this->combatantAAction->battlePhase == "defense") {
252             $first = "combatantA";
253         } elseif ($this->combatantBAction->battlePhase == "defense") {
254             $first = "combatantB";
255         }
256         $second = ($first == "combatantA" ? "combatantB" : "combatantA");
257         
258         // First action
259         $event = new BattleActionEvent($this, 
260                 $this->{$first}, $this->{$second},
261                 $this->{$first . "Action"}
262         );
263         $this->onBeforeAction($event);
264         $this->{$first . "Action"}->call("resolve", $this, $this->{$first}, $this->{$second});
265         $this->onAfterAction($event);
266         
267         // Second action
268         $event = new BattleActionEvent($this, 
269                 $this->{$second}, $this->{$first}, 
270                 $this->{$second . "Action"}
271         );
272         $this->onBeforeAction($event);
273         $this->{$second . "Action"}->call("resolve", $this, $this->{$second}, $this->{$first});
274         $this->onAfterAction($event);
275 
276         
277         // Clean up
278         $this->combatantAAction = null;
279         $this->combatantBAction = null;
280 
281         /*
282          * Check if fight is over
283          */
284         if($this->combatantA->hp <= 0 || $this->combatantB->hp <= 0) {
285             $this->stop();
286         } else {
287             $this->onAfterRound(new BattleEvent($this));
288 
289             $this->combatantA->save();
290             if($this->type == "pvp") {
291                 $this->combatantB->save();
292             }
293             $this->saveObjectState();
294         }
295     }

600     /**
601      * Prepares the object for hibernation
602      * Sets combatantA and B to null and removes inactive Battleeffects
603      * @return array array_keys((array)$this)
604      */
605     public function __sleep() {
606         if($this->type == "pvp") {
607             $this->combatantA = null;
608             $this->combatantB = null;
609         } else {
610             $this->combatantA = null;
611         }
612         
613         /**
614          * This is ugly, but you can't remove items from a CList while cycling 
615          * through the list with foreach. The iterator's $_i for the current 
616          * position is not updated. Apparently, this is how it should be. Weird.
617          */
618         $toDelete = array();
619         foreach($this->battleeffects as $battleeffect) {
620             if(!$battleeffect->active) {
621                 $toDelete[] = $battleeffect;
622             }
623         }
624         foreach($toDelete as $deletee) {
625             $this->battleeffects->remove($deletee);
626         }
627         
628         return parent::__sleep();
629     }
